🚀 快速开始

# 安装
git clone https://github.com/lxztry/lxzclaw.git
cd lxzclaw
npm install

# 配置 .env
echo "LXZ_LLM_PROVIDER=glm" > .env
echo "LXZ_GLM_API_KEY=你的密钥" >> .env

# 运行
npm run dev
